
local interfaceVersion = select(4, GetBuildInfo())

-- Global variables to hold our tracked focus data
MyFocusAddon_CurrentToken = nil
MyFocusAddon_CurrentGUID = nil
MyFocusAddon_CurrentName = nil

-- Opret rammens globale variabel med det samme under indlæsning
MyCustomFocusEraFrame = CreateFrame("Button", "MyCustomFocusEraFrame", UIParent, "SecureUnitButtonTemplate, BackdropTemplate")
local FocusFrame = MyCustomFocusEraFrame

-- Safe print load verification
local interfaceVersion = select(4, GetBuildInfo())
print("|cff00ff00[Z-Perl Focus Bridge]|r Addon loaded successfully on client version: " .. interfaceVersion)

-- DESIGN RAMMEN LYNHURTIGT UNDER LOADING
FocusFrame:SetSize(210, 48)
FocusFrame:SetPoint("CENTER", UIParent, "CENTER", 0, -100)
FocusFrame:SetMovable(true)
FocusFrame:EnableMouse(true)
FocusFrame:RegisterForClicks("AnyUp")
FocusFrame:Hide()

-- Ægte Z-Perl stil: Sort baggrund med skarp kant
FocusFrame:SetBackdrop({
    bgFile = "Interface\\Buttons\\WHITE8X8",
    edgeFile = "Interface\\Tooltips\\UI-Tooltip-Border",
    tile = true, tileSize = 8, edgeSize = 12,
    insets = { left = 2, right = 2, top = 2, bottom = 2 }
})
FocusFrame:SetBackdropColor(0, 0, 0, 0.85)
FocusFrame:SetBackdropBorderColor(0.4, 0.4, 0.4, 1)

-- Gør rammen flytbar uden for kamp
FocusFrame:RegisterForDrag("LeftButton")
FocusFrame:SetScript("OnDragStart", function(self) if not InCombatLockdown() then self:StartMoving() end end)
FocusFrame:SetScript("OnDragStop", function(self) self:StopMovingOrSizing() end)

-- 3D Portræt
FocusFrame.portrait = CreateFrame("PlayerModel", nil, FocusFrame)
FocusFrame.portrait:SetSize(40, 38)
FocusFrame.portrait:SetPoint("TOPLEFT", FocusFrame, "TOPLEFT", 6, -5)

local portBG = FocusFrame:CreateTexture(nil, "BACKGROUND")
portBG:SetAllPoints(FocusFrame.portrait)
portBG:SetColorTexture(0.05, 0.05, 0.05, 1)

-- Tekstfelter
FocusFrame.nameText = FocusFrame:CreateFontString(nil, "OVERLAY", "GameFontNormalSmall")
FocusFrame.nameText:SetPoint("TOPLEFT", FocusFrame, "TOPLEFT", 52, -6)
FocusFrame.nameText:SetJustifyH("LEFT")

FocusFrame.levelText = FocusFrame:CreateFontString(nil, "OVERLAY", "GameFontHighlightSmall")
FocusFrame.levelText:SetPoint("TOPRIGHT", FocusFrame, "TOPRIGHT", -10, -6)
FocusFrame.levelText:SetJustifyH("RIGHT")

-- Health Bar
FocusFrame.hpBar = CreateFrame("StatusBar", nil, FocusFrame)
FocusFrame.hpBar:SetSize(148, 14)
FocusFrame.hpBar:SetPoint("TOPLEFT", FocusFrame, "TOPLEFT", 52, -18)
FocusFrame.hpBar:SetStatusBarTexture("Interface\\TargetingFrame\\UI-StatusBar")
FocusFrame.hpBar:SetStatusBarColor(0, 1, 0)

FocusFrame.hpText = FocusFrame.hpBar:CreateFontString(nil, "OVERLAY", "GameFontHighlightSmall")
FocusFrame.hpText:SetPoint("CENTER", FocusFrame.hpBar, "CENTER", 0, 0)

-- Mana Bar
FocusFrame.manaBar = CreateFrame("StatusBar", nil, FocusFrame)
FocusFrame.manaBar:SetSize(148, 6)
FocusFrame.manaBar:SetPoint("TOPLEFT", FocusFrame.hpBar, "BOTTOMLEFT", 0, -2)
FocusFrame.manaBar:SetStatusBarTexture("Interface\\TargetingFrame\\UI-StatusBar")
FocusFrame.manaBar:SetStatusBarColor(0, 0, 1)

-- DEN UNIVERSELLE REGISTRERINGSTABEL (Clique læser denne uanset hvornår den loader)
ClickCastFrames = ClickCastFrames or {}
ClickCastFrames[FocusFrame] = true


-- Heartbeat Update Loop (10 gange i sekundet)
FocusFrame.timeSinceLastUpdate = 0
FocusFrame.lastRenderedGUID = nil

FocusFrame:SetScript("OnUpdate", function(self, elapsed)
    if not MyFocusAddon_CurrentToken then 
        self:Hide()
        return 
    end

    self.timeSinceLastUpdate = self.timeSinceLastUpdate + elapsed
    if self.timeSinceLastUpdate >= 0.1 then
        local token = MyFocusAddon_CurrentToken

        -- DYNAMIC TOKEN VERIFICATION (Sikrer holdkammerater ved gruppeskift)
        if not UnitExists(token) or UnitGUID(token) ~= MyFocusAddon_CurrentGUID then
            local remapped = false
            if IsInRaid() then
                for i = 1, 40 do
                    if UnitGUID("raid"..i) == MyFocusAddon_CurrentGUID then
                        MyFocusAddon_CurrentToken = "raid"..i
                        token = "raid"..i
                        remapped = true
                        break
                    end
                end
            else
                if UnitGUID("player") == MyFocusAddon_CurrentGUID then
                    MyFocusAddon_CurrentToken = "player"
                    token = "player"
                    remapped = true
                else
                    for i = 1, 4 do
                        if UnitGUID("party"..i) == MyFocusAddon_CurrentGUID then
                            MyFocusAddon_CurrentToken = "party"..i
                            token = "party"..i
                            remapped = true
                            break
                        end
                    end
                end
            end
            if not remapped then return end
        end

        if not UnitExists(token) then return end

        -- OPDATER VÆRDIER PÅ SKÆRMEN
        self.nameText:SetText(MyFocusAddon_CurrentName)
        local level = UnitLevel(token)
        self.levelText:SetText(level and level > 0 and "Lvl " .. level or "")

        local currentHP = UnitHealth(token)
        local maxHP = UnitHealthMax(token)
        self.hpBar:SetMinMaxValues(0, maxHP)
        self.hpBar:SetValue(currentHP)
        self.hpText:SetText(currentHP .. " / " .. maxHP)

        local _, class = UnitClass(token)
        if class and RAID_CLASS_COLORS[class] then
            local c = RAID_CLASS_COLORS[class]
            self.hpBar:SetStatusBarColor(c.r, c.g, c.b)
        end

        local powerType = UnitPowerType(token)
        local currentPower = UnitPower(token, powerType)
        local maxPower = UnitPowerMax(token, powerType)
        self.manaBar:SetMinMaxValues(0, maxPower)
        self.manaBar:SetValue(currentPower)

        if powerType == 0 then self.manaBar:SetStatusBarColor(0, 0.4, 1)
        elseif powerType == 1 then self.manaBar:SetStatusBarColor(1, 0, 0)
        elseif powerType == 3 then self.manaBar:SetStatusBarColor(1, 1, 0)
        else self.manaBar:SetStatusBarColor(0, 0.8, 0.8) end

        if self.portrait and self.lastRenderedGUID ~= MyFocusAddon_CurrentGUID then
            self.portrait:SetUnit(token)
            self.portrait:SetCamera(0)
            self.lastRenderedGUID = MyFocusAddon_CurrentGUID
        end

        self:Show()
        self.timeSinceLastUpdate = 0
    end
end)

-- FUNKTION TIL AT SÆTTE FOKUS
function MyFocusAddon_SetGroupFocus(unitToken)
    if InCombatLockdown() then 
        print("|cffff0000[Focus Era]|r Cannot change focus while in combat!")
        return 
    end

    if unitToken and FocusFrame then
        MyFocusAddon_CurrentToken = unitToken
        MyFocusAddon_CurrentGUID = UnitGUID(unitToken)
        MyFocusAddon_CurrentName = UnitName(unitToken)

        print("|cff00ff00[Focus Era]|r Focus target set to: " .. MyFocusAddon_CurrentName .. " (" .. unitToken .. ")")
        
        -- Det rå token kobles pivåbent til rammens secure attributes
        FocusFrame:SetAttribute("unit", unitToken)
        FocusFrame:SetAttribute("*unit", unitToken)
        
        FocusFrame.lastRenderedGUID = nil
        FocusFrame:Show()
    end
end

-- THE SLASH COMMANDS
SLASH_FOKUS1 = "/fokus"
SlashCmdList["FOKUS"] = function(msg)
    if InCombatLockdown() then
        print("|cffff0000[Focus Era]|r Cannot change focus during combat lockdown!")
        return
    end

    if not UnitExists("target") then
        print("|cffffaa00[Focus Era]|r You do not have a target selected.")
        return
    end

    local targetGUID = UnitGUID("target")
    local foundToken = nil

    if IsInRaid() then
        for i = 1, 40 do
            local token = "raid" .. i
            if UnitExists(token) and UnitGUID(token) == targetGUID then
                foundToken = token
                break
            end
        end
    else
        if UnitGUID("player") == targetGUID then
            foundToken = "player"
        else
            for i = 1, 4 do
                local token = "party" .. i
                if UnitExists(token) and UnitGUID(token) == targetGUID then
                    foundToken = token
                    break
                end
            end
        end
    end

    if foundToken then
        MyFocusAddon_SetGroupFocus(foundToken)
    else
        print("|cffff0000[Focus Era]|r Error: Target must be yourself or a member of your party/raid.")
    end
end

SLASH_CLEARFOKUS1 = "/clearfokus"
SlashCmdList["CLEARFOKUS"] = function(msg)
    if InCombatLockdown() then
        print("|cffff0000[Focus Era]|r Cannot clear focus during combat lockdown!")
        return
    end

    MyFocusAddon_CurrentToken = nil
    MyFocusAddon_CurrentGUID = nil
    MyFocusAddon_CurrentName = nil

    if FocusFrame then
        FocusFrame:SetAttribute("unit", nil)
        FocusFrame:SetAttribute("*unit", nil)
        FocusFrame.lastRenderedGUID = nil
        FocusFrame:Hide()
    end
    print("|cff00ff00[Focus Era]|r Focus cleared.")
end
